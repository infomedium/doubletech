    Linux Commands

    Operating System

    Hardware

    CPU

    # Show CPU information
lscpu

# Get information from file
cat /proc/cpuinfo

# Show number of processors
nproc

    Memory

    # Display amount of free and used memory in the system
free -g -h -t

# Get information from file
cat /proc/meminfo

    Disk

    # Get disk-level space usage [High Level]
df -h --total

# Get details on specific mount point
df -h /

# Get folder-level space usage [Low Level]
du -sh /tmp
# 676M    /tmp

    General

    # Show system information
# -a: all
uname -a

# Get machine hardware name
arch

    Tasks

    # Display information about the CPU architecture
top

# interactive process viewer
htop

    User Management

    # List all users
cat /etc/passwd

    File Operations

    # Delete folder recursively
# -r :recursive
# -f :force (no prompt)
rm -r temp1/
rm -rf temp1/
