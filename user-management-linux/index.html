    Linux User Management

    List all users and groups

    # List all users

cat /etc/passwd
# ...
# user1:x:1001:1001:,,,:/home/user1:/bin/bash

getent passwd
# ...
# user1:x:1001:1001:,,,:/home/user1:/bin/bash

    # List all groups

cat /etc/group
# ...
# user1:x:1001:

getent group
# ...
# user1:x:1001:

    Get user details and group details

    # Get basic details of user

id user1
# uid=1001(user1) gid=1001(user1) groups=1001(user1)

# Get basic details of account
getent passwd user1
# user1:x:1001:1001:,,,:/home/user1:/bin/bash

cat /etc/passwd | grep user1
# user1:x:1001:1001:,,,:/home/user1:/bin/bash

    # Get group details of user

groups user1
# user1 : user1

getent group user1
# user1:x:1001:

cat /etc/group | grep user1
# user1:x:1001:

    List all members of a group

    # Get all members belonging to a group

sudo apt install members # If not already installed

members adm
# syslog ubuntu

    Create a new account

    Create user account with same group

    Username Group Home directory Type
    user1 user1 /home/user1 User account 
    sudo useradd --home-dir /home/user1  user1

# Get basic details of account
getent passwd user1
# user1:x:1001:1001::/home/user1:/bin/sh

# Get home directory 
echo ~user1
# /home/user1

# Get group details of account
groups user1
# user1 : user1

    NOTE: If folders are not created, create them manually. 

    sudo mkdir /home/user1

# Change ownership
sudo chown user1:user1 /home/user1

    Create user account with different group

    Username Group Home directory Type
    user1 group1 /home/user1 User account 
    # Create new group
sudo groupadd group1

# Get group details
getent group group1
# group1:x:1002:

    # Add new user
sudo useradd --home-dir /home/user1 --gid group1 user1

# Get basic details of account
getent passwd user1
# user1:x:1001:1001::/home/user1:/bin/sh

# Get home directory 
echo ~user1
# /home/user1

# Get group details of account
groups user1
# user1 : group1

    Create service account

    Username Group Home directory Type
    system1 system1 Service account 
    sudo useradd  --system --no-create-home system1

# Get basic details of account
getent passwd system1
# system1:x:998:998::/home/system1:/bin/sh

# Get home directory 
echo ~user1
# /home/user1

# Get group details of account
groups system1
# system1 : system1

    Set password

    sudo passwd user1

# New password:
# Retype new password:
# passwd: password updated successfully

    Delete existing account

    sudo userdel user1

sudo userdel system1
